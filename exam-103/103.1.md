103.1 Work on the command line

```
echo $SHELL
cat /etc/shells
cat /etc/passwd

chsh -l
chsh <user>

YII_ENV=dev
echo $YII_ENV

sh
echo $YII_ENV

ps --forest
exit

export YII_ENV
sh
echo $YII_ENV

env YII_ENV=test ./app.sh

set +-o
set +o history (disable)
set -o history (enable)

history
!<n> executa o comando do histórico
!? executa o comando com o o padrão escrito
!! executa o último programa
```
